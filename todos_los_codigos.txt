/**
 * Muestra u oculta un elemento del DOM.
 * @param {string} id - ID del elemento
 * @param {boolean} visible - true para mostrar, false para ocultar
 */
export function mostrarElemento(id, visible) {
  const el = document.getElementById(id);
  if (!el) return;
  if (visible) {
    el.classList.remove("d-none");
  } else {
    el.classList.add("d-none");
  }
}

  const el = document.getElementById(id);

}

/**
 * Renderiza la lista de jugadores en el DOM.
 * Crea elementos li con botones de editar y eliminar para cada jugador.
 * @param {Array} jugadores - Lista de objetos jugador
 * @param {Function} onEditar - Callback al hacer clic en editar
 * @param {Function} onEliminar - Callback al hacer clic en eliminar
 */
export function renderizarListaJugadores(jugadores, onEditar, onEliminar) {
  const lista = document.getElementById("listaJugadores");
  const sinJugadores = document.getElementById("mensajeSinJugadores");
  const contador = document.getElementById("contadorJugadores");

  if (!lista) return;

  /* Limpiar lista actual */
  lista.innerHTML = "";

  /* Actualizar contador */
  if (contador) contador.textContent = jugadores.length;

  /* Mostrar mensaje si no hay jugadores */
  if (jugadores.length === 0) {
    if (sinJugadores) sinJugadores.classList.remove("d-none");
    return;
  }

  if (sinJugadores) sinJugadores.classList.add("d-none");

  /* Ordenar por puntaje descendente */
  const ordenados = [...jugadores].sort((a, b) => b.puntaje - a.puntaje);

  ordenados.forEach((jugador, indice) => {
    const li = document.createElement("li");
    li.className =
      "list-group-item d-flex justify-content-between align-items-start py-2 px-2";
    li.dataset.id = jugador.id;

    li.innerHTML = `
      <div>
        <span class="fw-semibold">${indice + 1}. ${jugador.nombre}</span>
        <br/>
        <small class="text-muted">
          Puntaje: <strong>${jugador.puntaje}</strong> |
          Modo: ${jugador.modo} | ${jugador.fecha}
        </small>
      </div>
      <div class="d-flex gap-1 ms-2">
        <button class="btn btn-outline-primary btn-sm btnEditar" data-id="${jugador.id}">
          <i class="bi bi-pencil"></i>
        </button>
        <button class="btn btn-outline-danger btn-sm btnEliminar" data-id="${jugador.id}">
          <i class="bi bi-trash"></i>
        </button>
      </div>
    `;

    /* Eventos de editar y eliminar sobre cada Ã­tem */
    li.querySelector(".btnEditar").addEventListener("click", () => {
      onEditar(jugador);
    });
    li.querySelector(".btnEliminar").addEventListener("click", () => {
      onEliminar(jugador.id);
    });

    lista.appendChild(li);
  });
}

/**
 * Descarga un contenido como archivo .txt en el navegador del usuario.
 * Usa Blob + URL.createObjectURL() + elemento <a> dinÃ¡mico.
 * @param {string} contenido - Texto a guardar en el archivo
 * @param {string} nombreArchivo - Nombre sugerido para el archivo
 */
export function descargarTxt(contenido, nombreArchivo) {
  const blob = new Blob([contenido], { type: "text/plain;charset=utf-8" });
  const url = URL.createObjectURL(blob);
  const enlace = document.createElement("a");
  enlace.href = url;
  enlace.download = nombreArchivo;
  document.body.appendChild(enlace);
  enlace.click();
  document.body.removeChild(enlace);
  URL.revokeObjectURL(url);
}

 * LÃ³gica completa del juego Snake con sistema de niveles y velocidad progresiva.

/** TamaÃ±o de cada celda en pÃ­xeles */
const TAMANO_CELDA = 20;

/** Cantidad de celdas en cada dimensiÃ³n */
const CELDAS = 20;

/**
 * ConfiguraciÃ³n de niveles.
 * Nivel 7 es infinito: velocidad fija en 5, nunca cambia.
 */
const CONFIG_NIVELES = [
  { nivel: 1, puntajeMin: 0,    puntajeMax: 49,  velocidad: 1 },
  { nivel: 2, puntajeMin: 50,  puntajeMax: 99,  velocidad: 2 },
  { nivel: 3, puntajeMin: 100,  puntajeMax: 149,  velocidad: 3 },
  { nivel: 4, puntajeMin: 150,  puntajeMax: 199,  velocidad: 4 },
  { nivel: 5, puntajeMin: 200,  puntajeMax: 249,  velocidad: 5 },
  { nivel: 6, puntajeMin: 250, puntajeMax: 349, velocidad: 6 },
  { nivel: 7, puntajeMin: 350, puntajeMax: Infinity, velocidad: 5 }
];

/**
 * Determina el nivel correspondiente a un puntaje dado.
 * @param {number} puntaje
 * @returns {{ nivel: number, velocidad: number }}
 */
function obtenerNivelPorPuntaje(puntaje) {
  for (let i = CONFIG_NIVELES.length - 1; i >= 0; i--) {
    if (puntaje >= CONFIG_NIVELES[i].puntajeMin) {
      return {
        nivel: CONFIG_NIVELES[i].nivel,
        velocidad: CONFIG_NIVELES[i].velocidad
      };
    }
  }
  return { nivel: 1, velocidad: 1 };
}

/**
 * Convierte el nivel de velocidad al intervalo en ms del setInterval.
 * Mayor nivel â†’ intervalo menor â†’ mÃ¡s rÃ¡pido.
 * Nivel 1 â†’ 200ms, Nivel 6 â†’ 100ms, Nivel 7 â†’ 120ms (velocidad 5)
 * @param {number} velocidadNivel - 1 a 6
 * @returns {number} milisegundos
 */
function velocidadAIntervalo(velocidadNivel) {
  return Math.max(80, 220 - velocidadNivel * 20);
}

/**
 * Crea el estado inicial de una serpiente.
 */
function crearSerpiente(xInicio, yInicio, color, colorBorde) {
  return {
    cuerpo: [
      { x: xInicio,     y: yInicio },
      { x: xInicio - 1, y: yInicio },
      { x: xInicio - 2, y: yInicio }
    ],
    direccion:        { x: 1, y: 0 },
    proximaDireccion: { x: 1, y: 0 },
    color,
    colorBorde,
    viva:    true,
    puntaje: 0
  };
}

/**
 * Genera una posiciÃ³n aleatoria para la comida,
 * evitando las posiciones ocupadas por las serpientes.
 */
function generarComida(serpientes) {
  let posicion;
  let ocupada = true;
  while (ocupada) {
    posicion = {
      x: Math.floor(Math.random() * CELDAS),
      y: Math.floor(Math.random() * CELDAS)
    };
    ocupada = serpientes.some((s) =>
      s.cuerpo.some((seg) => seg.x === posicion.x && seg.y === posicion.y)
    );
  }
  return posicion;
}

/**
 * Clase principal que encapsula el juego Snake.
 * Ahora incluye sistema de niveles progresivos con velocidad automÃ¡tica.
 */
export class JuegoSnake {
  /**
   * @param {HTMLCanvasElement} canvas
   * @param {Object} config
   * @param {string}   config.modo         - '1jugador' | '2jugadores'
   * @param {Function} config.onPuntaje    - (p1, p2, nivel, velocidad)
   * @param {Function} config.onFinJuego   - (p1, p2, nivel)
   * @param {Function} config.onCambioNivel - (nuevoNivel) â€” para notificaciÃ³n visual
   */
  constructor(canvas, config) {
    this.canvas  = canvas;
    this.ctx     = canvas.getContext('2d');
    this.config  = config;
    this.intervalo = null;
    this.pausado   = false;

    /** Nivel actual de juego (1-7) */
    this.nivelActual = 1;

    /** Velocidad actual (1-6, nivel 7 usa velocidad 5) */
    this.velocidadActual = 1;

    /** Intervalo en ms del bucle (calculado desde velocidadActual) */
    this.intervaloMs = velocidadAIntervalo(1);

    /* Serpientes */
    this.serpiente1 = crearSerpiente(10, 10, '#00ff88', '#00cc66');
    this.serpiente2 = config.modo === '2jugadores'
      ? crearSerpiente(5, 5, '#4488ff', '#2266cc')
      : null;

    const serpientes = this.serpiente2
      ? [this.serpiente1, this.serpiente2]
      : [this.serpiente1];

    this.comida = generarComida(serpientes);
  }

  /**
   * Establece la velocidad inicial desde el selector de velocidad del formulario.
   * Solo se aplica si el nivel calculado por puntaje no lo supera.
   * @param {number} nivel - 1 a 10 (valor del range)
   */
  establecerVelocidad(nivel) {
    /* Usamos el valor del range como velocidad inicial base,
       pero lo limitamos al rango 1-6 para compatibilidad con el sistema */
    const velBase = Math.min(6, Math.max(1, nivel));
    this.velocidadActual = velBase;
    this.intervaloMs     = velocidadAIntervalo(velBase);
  }

  /**
   * Verifica si el puntaje actual requiere cambio de nivel.
   * Si cambia, reinicia el intervalo con la nueva velocidad.
   * @param {number} puntajeJ1
   * @param {number} puntajeJ2
   */
  _actualizarNivel(puntajeJ1, puntajeJ2) {
    /* En modo 2 jugadores, el nivel se calcula con el puntaje mayor */
    const puntajeReferencia = Math.max(puntajeJ1, puntajeJ2);
    const { nivel, velocidad } = obtenerNivelPorPuntaje(puntajeReferencia);

    if (nivel !== this.nivelActual) {
      const nivelAnterior = this.nivelActual;
      this.nivelActual     = nivel;
      this.velocidadActual = velocidad;
      this.intervaloMs     = velocidadAIntervalo(velocidad);

      /* Reiniciar el intervalo con la nueva velocidad */
      if (this.intervalo) {
        clearInterval(this.intervalo);
        this.intervalo = setInterval(() => this.actualizar(), this.intervaloMs);
      }

      /* Notificar cambio de nivel al exterior */
      if (this.config.onCambioNivel) {
        this.config.onCambioNivel(nivel, nivelAnterior);
      }
    }
  }

  /**
   * Inicia el bucle principal del juego.
   */
  iniciar() {
    if (this.intervalo) clearInterval(this.intervalo);
    this.intervalo = setInterval(() => this.actualizar(), this.intervaloMs);
  }

  /**
   * Alterna entre pausado y activo.
   * @returns {boolean} true si quedÃ³ pausado
   */
  togglePausa() {
    if (this.pausado) {
      this.intervalo = setInterval(() => this.actualizar(), this.intervaloMs);
      this.pausado = false;
    } else {
      clearInterval(this.intervalo);
      this.pausado = true;
    }
    return this.pausado;
  }

  /**
   * Detiene el juego limpiando el intervalo.
   */
  detener() {
    if (this.intervalo) clearInterval(this.intervalo);
  }

  /**
   * Cambia la direcciÃ³n de la serpiente 1.
   * Impide el giro de 180 grados.
   */
  cambiarDireccionJ1(nuevaDireccion) {
    const dir = this.serpiente1.direccion;
    if (nuevaDireccion.x !== -dir.x || nuevaDireccion.y !== -dir.y) {
      this.serpiente1.proximaDireccion = nuevaDireccion;
    }
  }

  /**
   * Cambia la direcciÃ³n de la serpiente 2.
   */
  cambiarDireccionJ2(nuevaDireccion) {
    if (!this.serpiente2) return;
    const dir = this.serpiente2.direccion;
    if (nuevaDireccion.x !== -dir.x || nuevaDireccion.y !== -dir.y) {
      this.serpiente2.proximaDireccion = nuevaDireccion;
    }
  }

  /**
   * Mueve una serpiente en su direcciÃ³n actual.
   */
  moverSerpiente(serpiente, todasSerpientes) {
    if (!serpiente.viva) return;

    serpiente.direccion = { ...serpiente.proximaDireccion };

    const cabeza = serpiente.cuerpo[0];
    const nuevaCabeza = {
      x: cabeza.x + serpiente.direccion.x,
      y: cabeza.y + serpiente.direccion.y
    };

    /* ColisiÃ³n con paredes */
    if (
      nuevaCabeza.x < 0 || nuevaCabeza.x >= CELDAS ||
      nuevaCabeza.y < 0 || nuevaCabeza.y >= CELDAS
    ) {
      serpiente.viva = false;
      return;
    }

    /* ColisiÃ³n con cualquier cuerpo */
    for (const s of todasSerpientes) {
      for (const seg of s.cuerpo) {
        if (nuevaCabeza.x === seg.x && nuevaCabeza.y === seg.y) {
          serpiente.viva = false;
          return;
        }
      }
    }

    serpiente.cuerpo.unshift(nuevaCabeza);

    /* Verificar si comiÃ³ la comida */
    if (nuevaCabeza.x === this.comida.x && nuevaCabeza.y === this.comida.y) {
      serpiente.puntaje += 10;

      const serpientes = this.serpiente2
        ? [this.serpiente1, this.serpiente2]
        : [this.serpiente1];
      this.comida = generarComida(serpientes);

      /* Notificar cambio de puntaje con nivel y velocidad */
      this.config.onPuntaje(
        this.serpiente1.puntaje,
        this.serpiente2 ? this.serpiente2.puntaje : 0,
        this.nivelActual,
        this.velocidadActual
      );

      /* Verificar si corresponde cambiar de nivel */
      this._actualizarNivel(
        this.serpiente1.puntaje,
        this.serpiente2 ? this.serpiente2.puntaje : 0
      );
    } else {
      serpiente.cuerpo.pop();
    }
  }

  /**
   * Ciclo principal de actualizaciÃ³n.
   */
  actualizar() {
    const serpientes = this.serpiente2
      ? [this.serpiente1, this.serpiente2]
      : [this.serpiente1];

    for (const s of serpientes) {
      this.moverSerpiente(s, serpientes);
    }

    const juegoTerminado = serpientes.some((s) => !s.viva);
    if (juegoTerminado) {
      this.detener();
      this.config.onFinJuego(
        this.serpiente1.puntaje,
        this.serpiente2 ? this.serpiente2.puntaje : 0,
        this.nivelActual
      );
    }

    this.renderizar();
  }

  /**
   * Renderiza el estado actual del juego en el canvas.
   */
  renderizar() {
    const ctx  = this.ctx;
    const ancho = this.canvas.width;
    const alto  = this.canvas.height;

    ctx.fillStyle = '#1a1a2e';
    ctx.fillRect(0, 0, ancho, alto);

    /* Grilla sutil */
    ctx.strokeStyle = 'rgba(255,255,255,0.03)';
    ctx.lineWidth   = 0.5;
    for (let i = 0; i <= CELDAS; i++) {
      ctx.beginPath(); ctx.moveTo(i * TAMANO_CELDA, 0); ctx.lineTo(i * TAMANO_CELDA, alto); ctx.stroke();
      ctx.beginPath(); ctx.moveTo(0, i * TAMANO_CELDA); ctx.lineTo(ancho, i * TAMANO_CELDA); ctx.stroke();
    }

    /* Comida */
    ctx.fillStyle  = '#ff4466';
    ctx.shadowColor = '#ff4466';
    ctx.shadowBlur  = 10;
    ctx.beginPath();
    ctx.arc(
      this.comida.x * TAMANO_CELDA + TAMANO_CELDA / 2,
      this.comida.y * TAMANO_CELDA + TAMANO_CELDA / 2,
      TAMANO_CELDA / 2 - 2, 0, Math.PI * 2
    );
    ctx.fill();
    ctx.shadowBlur = 0;

    /* Serpientes */
    const serpientes = this.serpiente2
      ? [this.serpiente1, this.serpiente2]
      : [this.serpiente1];

    for (const s of serpientes) {
      if (!s.viva) continue;
      s.cuerpo.forEach((seg, idx) => {
        ctx.fillStyle   = idx === 0 ? s.color : s.color + 'bb';
        ctx.strokeStyle = s.colorBorde;
        ctx.lineWidth   = 1;
        ctx.fillRect(
          seg.x * TAMANO_CELDA + 1, seg.y * TAMANO_CELDA + 1,
          TAMANO_CELDA - 2, TAMANO_CELDA - 2
        );
        ctx.strokeRect(
          seg.x * TAMANO_CELDA + 1, seg.y * TAMANO_CELDA + 1,
          TAMANO_CELDA - 2, TAMANO_CELDA - 2
        );
      });
    }
  }
}

const MAPA_DIR_J1 = {
  arriba:    { x: 0,  y: -1 },
  abajo:     { x: 0,  y:  1 },
  izquierda: { x: -1, y:  0 },
  derecha:   { x: 1,  y:  0 }
};

const MAPA_DIR_J2 = {
  arriba:    { x: 0,  y: -1 },
  abajo:     { x: 0,  y:  1 },
  izquierda: { x: -1, y:  0 },
  derecha:   { x: 1,  y:  0 }
};

function registrarBotonTactil(idBoton, jugador, dir) {
  const btn = document.getElementById(idBoton);
  if (!btn) return;

  const manejarPresion = (e) => {
    e.preventDefault();
    if (!juegoActual || !partidaActiva) return;

    if (jugador === 1) {
      const direccion = MAPA_DIR_J1[dir];
      if (direccion) juegoActual.cambiarDireccionJ1(direccion);
    } else if (jugador === 2) {
      const direccion = MAPA_DIR_J2[dir];
      if (direccion) juegoActual.cambiarDireccionJ2(direccion);
    }
  };

  btn.addEventListener("mousedown", manejarPresion);
  btn.addEventListener("touchstart", manejarPresion, { passive: false });
}

  registrarBotonTactil("j2Izquierda", 2, "izquierda");
  registrarBotonTactil("j2Derecha",   2, "derecha");
}

function inicializarEventos() {

  /* â”€â”€ Selector de modo â”€â”€ */
  document.getElementById("selectModo").addEventListener("change", (e) => {
    modoActual = e.target.value;
    const es2J = modoActual === "2jugadores";
    mostrarElemento("contenedorJ2", es2J);
    mostrarElemento("controlesJ2Info", es2J);
    mostrarElemento("filaPuntajeJ2", es2J);
  });

  /* â”€â”€ Velocidad en tiempo real â”€â”€ */
  document.getElementById("rangeVelocidad").addEventListener("input", (e) => {
    establecerTexto("labelVelocidad", e.target.value);
  });

  /* â”€â”€ ValidaciÃ³n blur J1 â”€â”€ */
  document.getElementById("inputNombreJ1").addEventListener("blur", (e) => {
    toggleError("errorNombreJ1", !validarNombre(e.target.value.trim()));
  });

  /* â”€â”€ ValidaciÃ³n blur J2 â”€â”€ */
  document.getElementById("inputNombreJ2").addEventListener("blur", (e) => {
    if (modoActual !== "2jugadores") return;
    toggleError("errorNombreJ2", !validarNombre(e.target.value.trim()));
  });

  /* â”€â”€ INICIAR JUEGO â”€â”€ */
  document.getElementById("btnIniciar").addEventListener("click", () => {
    const nombreJ1 = obtenerValor("inputNombreJ1");

    if (!validarNombre(nombreJ1)) {
      toggleError("errorNombreJ1", true);
      actualizarMensajeEstado("IngresÃ¡ un nombre vÃ¡lido para el Jugador 1.", "danger");
      return;
    }
    toggleError("errorNombreJ1", false);

    if (modoActual === "2jugadores") {
      const nombreJ2 = obtenerValor("inputNombreJ2");
      if (!validarNombre(nombreJ2)) {
        toggleError("errorNombreJ2", true);
        actualizarMensajeEstado("IngresÃ¡ un nombre vÃ¡lido para el Jugador 2.", "danger");
        return;
      }
      toggleError("errorNombreJ2", false);
    }

    if (juegoActual) juegoActual.detener();

    /* El canvas ya estÃ¡ en contenedorCanvasConfig; lo usamos directamente */
    const canvas = document.getElementById("canvasJuego");

    const nombreJ1Real = obtenerValor("inputNombreJ1") || "Jugador 1";
    const nombreJ2Real = modoActual === "2jugadores"
      ? (obtenerValor("inputNombreJ2") || "Jugador 2")
      : "Jugador 2";

    establecerTexto("marcadorNombreJ1", nombreJ1Real);
    establecerTexto("marcadorNombreJ2", nombreJ2Real);
    establecerTexto("tituloControlesJ1", nombreJ1Real);
    establecerTexto("tituloControlesJ2", nombreJ2Real);

    juegoActual = new JuegoSnake(canvas, {
      modo: modoActual,

      onPuntaje: (p1, p2, nivel, velocidadNivel) => {
        establecerTexto("puntajeJ1", p1);
        if (modoActual === "2jugadores") establecerTexto("puntajeJ2", p2);
        actualizarMarcadorExpandido(p1, p2, nivel, velocidadNivel);
      },

      onCambioNivel: (nuevoNivel) => {
        mostrarNotificacionNivel(nuevoNivel);
      },

      onFinJuego: async (p1, p2, nivel) => {
        const nJ1 = obtenerValor("inputNombreJ1");
        await guardarJugador({ nombre: nJ1, puntaje: p1, modo: modoActual, nivel });

        if (modoActual === "2jugadores") {
          const nJ2 = obtenerValor("inputNombreJ2");
          await guardarJugador({ nombre: nJ2, puntaje: p2, modo: modoActual, nivel });
        }

        const msgExp = document.getElementById("mensajeGameOverExpandido");
        if (msgExp) {
          if (modoActual === "2jugadores") {
            const nJ2 = obtenerValor("inputNombreJ2");
            const ganador = p1 > p2 ? nJ1 : p2 > p1 ? nJ2 : "Empate";
            msgExp.textContent = ganador === "Empate"
              ? `Â¡Empate! Nivel alcanzado: ${nivel}`
              : `Â¡GanÃ³ ${ganador} con ${Math.max(p1, p2)} pts! Nivel: ${nivel}`;
          } else {
            msgExp.textContent = `Â¡Juego terminado! Puntaje: ${p1} | Nivel: ${nivel}`;
          }
        }
        mostrarElemento("filaGameOverExpandido", true);

        mostrarElemento("mensajeGameOver", true);
        mostrarElemento("btnIniciar", true);
        mostrarElemento("btnPausar", false);
        mostrarElemento("btnReiniciar", true);
        mostrarElemento("btnPausarExpandido", false);
        partidaActiva = false;

        await cargarYRenderizarJugadores();
      }
    });

    juegoActual.establecerVelocidad(parseInt(obtenerValor("rangeVelocidad"), 10));
    juegoActual.iniciar();
    partidaActiva = true;

    establecerTexto("labelJ1Puntaje", nombreJ1Real);
    if (modoActual === "2jugadores") establecerTexto("labelJ2Puntaje", nombreJ2Real);
    establecerTexto("puntajeJ1", "0");
    establecerTexto("puntajeJ2", "0");
    mostrarElemento("mensajeGameOver", false);
    mostrarElemento("btnIniciar", false);
    mostrarElemento("btnPausar", true);
    mostrarElemento("btnReiniciar", true);

    actualizarMarcadorExpandido(0, 0, 1, juegoActual.velocidadActual);

    /* Activar vista expandida: MUEVE el canvas fÃ­sicamente */
    activarVistaExpandida();
  });

  /* â”€â”€ PAUSAR / REANUDAR â”€â”€ */
  const manejarPausa = () => {
    if (!juegoActual) return;
    const pausado = juegoActual.togglePausa();
    const btnP  = document.getElementById("btnPausar");
    const btnPE = document.getElementById("btnPausarExpandido");
    if (pausado) {
      if (btnP)  btnP.innerHTML  = '<i class="bi bi-play-fill"></i> Reanudar';
      if (btnPE) btnPE.innerHTML = '<i class="bi bi-play-fill"></i> Reanudar';
      actualizarMensajeEstado("Juego pausado.", "warning");
    } else {
      if (btnP)  btnP.innerHTML  = '<i class="bi bi-pause-fill"></i> Pausar';
      if (btnPE) btnPE.innerHTML = '<i class="bi bi-pause-fill"></i> Pausar';
      actualizarMensajeEstado("Â¡Juego en curso!", "success");
    }
  };

  document.getElementById("btnPausar").addEventListener("click", manejarPausa);
  document.getElementById("btnPausarExpandido").addEventListener("click", manejarPausa);

  /* â”€â”€ REINICIAR â”€â”€ */
  const manejarReiniciar = () => {
    if (juegoActual) juegoActual.detener();
    juegoActual   = null;
    partidaActiva = false;

    establecerTexto("puntajeJ1", "0");
    establecerTexto("puntajeJ2", "0");
    mostrarElemento("mensajeGameOver", false);
    mostrarElemento("btnIniciar", true);
    mostrarElemento("btnPausar", false);
    mostrarElemento("btnReiniciar", false);
    actualizarMensajeEstado("ConfigurÃ¡ los jugadores e iniciÃ¡ la partida.", "info");

    volverAConfig();
  };

  document.getElementById("btnReiniciar").addEventListener("click", manejarReiniciar);
  document.getElementById("btnReiniciarExpandido").addEventListener("click", manejarReiniciar);

  /* â”€â”€ VOLVER A CONFIG â”€â”€ */
  document.getElementById("btnVolverConfig").addEventListener("click", volverAConfig);

  /* â”€â”€ TECLADO â”€â”€ */
  document.addEventListener("keydown", (e) => {
    if (!juegoActual || !partidaActiva) return;

    const mapaJ1 = {
      ArrowUp:    { x: 0,  y: -1 },
      ArrowDown:  { x: 0,  y:  1 },
      ArrowLeft:  { x: -1, y:  0 },
      ArrowRight: { x: 1,  y:  0 }
    };
    const mapaJ2 = {
      w: { x: 0,  y: -1 },
      s: { x: 0,  y:  1 },
      a: { x: -1, y:  0 },
      d: { x: 1,  y:  0 }
    };

    if (mapaJ1[e.key]) {
      e.preventDefault();
      juegoActual.cambiarDireccionJ1(mapaJ1[e.key]);
    }
    if (modoActual === "2jugadores" && mapaJ2[e.key]) {
      juegoActual.cambiarDireccionJ2(mapaJ2[e.key]);
    }
  });

  /* â”€â”€ GUARDAR EDICIÃ“N â”€â”€ */
  document.getElementById("btnGuardarEdicion").addEventListener("click", async () => {
    const id      = obtenerValor("editarId");
    const nombre  = obtenerValor("editarNombre");
    const puntaje = parseInt(obtenerValor("editarPuntaje"), 10);
    const modo    = document.getElementById("editarModo").value;

    if (!validarNombre(nombre)) {
      actualizarMensajeEstado("El nombre editado no es vÃ¡lido.", "danger");
      return;
    }
    await modificarJugador(id, { nombre, puntaje, modo });
    mostrarElemento("cardEditar", false);
    actualizarMensajeEstado("Jugador actualizado correctamente.", "success");
    await cargarYRenderizarJugadores();
  });

  /* â”€â”€ CANCELAR EDICIÃ“N â”€â”€ */
  document.getElementById("btnCancelarEdicion").addEventListener("click", () => {
    mostrarElemento("cardEditar", false);
  });

  /* â”€â”€ DESCARGAR TXT â”€â”€ */
  document.getElementById("btnDescargarTxt").addEventListener("click", async () => {
    const jugadores = await obtenerJugadores();
    const lineas = jugadores
      .sort((a, b) => b.puntaje - a.puntaje)
      .map((j, i) =>
        `${i + 1}. ${j.nombre} | Puntaje: ${j.puntaje} | Modo: ${j.modo} | Fecha: ${j.fecha}`
      ).join("\n");
    const contenido = `RANKING SNAKE\n${"=".repeat(40)}\n${lineas}\n${"=".repeat(40)}\nTotal jugadores: ${jugadores.length}`;
    descargarTxt(contenido, "ranking_snake.txt");
  });

  /* â”€â”€ TOGGLE TEMA â”€â”€ */
  document.getElementById("btnTema").addEventListener("click", toggleTema);

  /* â”€â”€ CONTROLES TÃCTILES â”€â”€ */
  inicializarControlesTactiles();
}

/**
 * Rutas Express para el CRUD completo de jugadores.
 * Persiste los datos en /data/jugadores.txt usando fs (ES Modules).
 *
 * Rutas disponibles:
 *  GET    /api/jugadores         â†’ obtener todos los jugadores
 *  POST   /api/jugadores         â†’ agregar un jugador
 *  PUT    /api/jugadores/:id     â†’ modificar un jugador
 *  DELETE /api/jugadores/:id     â†’ eliminar un jugador
 *  GET    /api/jugadores/archivo â†’ descargar el .txt
 */

import express from "express";
import fs from "fs";
import path from "path";
import { fileURLToPath } from "url";

const router = express.Router();
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

/** Ruta absoluta al archivo de persistencia */
const rutaArchivo = path.join(__dirname, "../data/jugadores.txt");

/**
 * Asegura que el directorio /data y el archivo existan.
 * Si no existen, los crea vacÃ­os.
 */
function asegurarArchivo() {
  const dir = path.join(__dirname, "../data");
  if (!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true });
  }
  if (!fs.existsSync(rutaArchivo)) {
    fs.writeFileSync(rutaArchivo, JSON.stringify([]), "utf-8");
  }
}

/**
 * Lee todos los jugadores del archivo .txt
 * @returns {Array} Lista de jugadores
 */
function leerJugadores() {
  asegurarArchivo();
  const contenido = fs.readFileSync(rutaArchivo, "utf-8");
  try {
    return JSON.parse(contenido);
  } catch {
    return [];
  }
}

/**
 * Escribe la lista de jugadores en el archivo .txt
 * @param {Array} jugadores
 */
function escribirJugadores(jugadores) {
  asegurarArchivo();
  fs.writeFileSync(rutaArchivo, JSON.stringify(jugadores, null, 2), "utf-8");
}

/* GET: obtener todos los jugadores */
router.get("/", (req, res) => {
  const jugadores = leerJugadores();
  res.json(jugadores);
});

/* GET: descargar el archivo .txt de jugadores */
router.get("/archivo", (req, res) => {
  asegurarArchivo();
  res.download(rutaArchivo, "jugadores.txt");
});

/**
 * POST: agregar un nuevo jugador
 * Recibe: { nombre, puntaje, modo }
 */
router.post("/", (req, res) => {
  const { nombre, puntaje, modo } = req.body;

  /* ValidaciÃ³n bÃ¡sica de datos recibidos */
  if (!nombre || puntaje === undefined || !modo) {
    return res.status(400).json({ error: "Datos incompletos" });
  }

  const jugadores = leerJugadores();

  /* Generar ID Ãºnico basado en timestamp */
  const nuevoJugador = {
    id: Date.now().toString(),
    nombre: nombre.trim(),
    puntaje: Number(puntaje),
    modo,
    fecha: new Date().toLocaleDateString("es-AR")
  };

  jugadores.push(nuevoJugador);
  escribirJugadores(jugadores);

  res.status(201).json(nuevoJugador);
});

/**
 * PUT: modificar un jugador existente por ID
 * Recibe: { nombre, puntaje, modo }
 */
router.put("/:id", (req, res) => {
  const { id } = req.params;
  const { nombre, puntaje, modo } = req.body;

  const jugadores = leerJugadores();
  const indice = jugadores.findIndex((j) => j.id === id);

  if (indice === -1) {
    return res.status(404).json({ error: "Jugador no encontrado" });
  }

  /* Actualizar campos si fueron enviados */
  if (nombre !== undefined) jugadores[indice].nombre = nombre.trim();
  if (puntaje !== undefined) jugadores[indice].puntaje = Number(puntaje);
  if (modo !== undefined) jugadores[indice].modo = modo;

  escribirJugadores(jugadores);
  res.json(jugadores[indice]);
});

/**
 * DELETE: eliminar un jugador por ID
 */
router.delete("/:id", (req, res) => {
  const { id } = req.params;
  const jugadores = leerJugadores();
  const nuevaLista = jugadores.filter((j) => j.id !== id);

  if (nuevaLista.length === jugadores.length) {
    return res.status(404).json({ error: "Jugador no encontrado" });
  }

  escribirJugadores(nuevaLista);
  res.json({ mensaje: "Jugador eliminado correctamente" });
});

export default router;

export function descargarTxt(contenido, nombreArchivo) {

    this._detectarColisiones();

    /* Notificar estado a la interfaz */
    this.config.onPuntaje(
      this.pacman1.puntaje,
      this.pacman2 ? this.pacman2.puntaje : 0,
      this.pacman1.vidas,
      this.pacman2 ? this.pacman2.vidas : 0,
      this.nivel
    );

    /* Â¿TerminÃ³ el juego? */
    const pm1Muerto = !this.pacman1.vivo;
    const pm2Muerto = this.pacman2 ? !this.pacman2.vivo : true;
    const juegoTerminado = this.pacman2
      ? (pm1Muerto && pm2Muerto)
      : pm1Muerto;

    if (juegoTerminado) {
      this.detener();
      this.renderizar();
      this.config.onFinJuego(
        this.pacman1.puntaje,
        this.pacman2 ? this.pacman2.puntaje : 0,
        this.nivel
      );
      return;
    }

    /* Â¿Se completÃ³ el nivel? */
    if (this.puntosRestantes <= 0) this._avanzarNivel();

    this.renderizar();
  }

  /* â”€â”€ Pac-Man: movimiento y consumo de puntos â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Mueve un Pac-Man en su direcciÃ³n actual.
   * Intenta aplicar la prÃ³xima direcciÃ³n si el camino estÃ¡ libre.
   * Consume puntos y power-ups al pasar sobre ellos.
   * @param {Object} pm - Objeto Pac-Man
   */
  _moverPacman(pm) {
    if (!pm.vivo) return;

    /* Intentar prÃ³xima direcciÃ³n */
    const { dc, df } = pm.proximaDireccion;
    if (esCeldaLibrePacman(this.mapa, pm.col + dc, pm.fila + df)) {
      pm.direccion = { ...pm.proximaDireccion };
    }

    /* Aplicar direcciÃ³n actual */
    const nc = pm.col + pm.direccion.dc;
    const nf = pm.fila + pm.direccion.df;

    if (esCeldaLibrePacman(this.mapa, nc, nf)) {
      pm.col = nc;
      pm.fila = nf;

      /* TÃºnel lateral */
      if (pm.col < 0)        pm.col = COLUMNAS - 1;
      if (pm.col >= COLUMNAS) pm.col = 0;
    }

    /* Consumir punto */
    const celda = this.mapa[pm.fila][pm.col];
    if (celda === 0) {
      pm.puntaje += 10;
      this.mapa[pm.fila][pm.col] = 2;
      this.puntosRestantes--;
      this.puntosComidos++;
      this.ticksSinComer = 0;          // reiniciar timer de inactividad
      this._verificarUmbralesFantasmas();
    } else if (celda === 3) {
      pm.puntaje += 50;
      this.mapa[pm.fila][pm.col] = 2;
      this.puntosRestantes--;
      this.puntosComidos++;
      this.ticksSinComer = 0;
      this._activarPoder(pm);
      this._verificarUmbralesFantasmas();
    }

    /* AnimaciÃ³n de la boca */
    if (pm.abriendoBoca) {
      pm.anguloApertura += 0.15;
      if (pm.anguloApertura >= 0.35) pm.abriendoBoca = false;
    } else {
      pm.anguloApertura -= 0.15;
      if (pm.anguloApertura <= 0)    pm.abriendoBoca = true;
    }

    /* Decrementar poder */
    if (pm.poderActivo) {
      pm.tiempoPoder--;
      if (pm.tiempoPoder <= 0) {
        pm.poderActivo = false;
        this._desactivarAsustar();
      }
    }
  }

  /* â”€â”€ Poder â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Activa el poder de Pac-Man: asusta a todos los fantasmas libres.
   * @param {Object} pm
   */
  _activarPoder(pm) {
    pm.poderActivo  = true;
    pm.tiempoPoder  = 50;
    for (const f of this.fantasmas) {
      if (f.estadoSalida === 'libre') {
        f.asustado      = true;
        f.color         = '#2222ff';
        f.tiempoAsustado = 50;
      }
    }
  }

  /** Desactiva el estado asustado de todos los fantasmas. */
  _desactivarAsustar() {
    for (const f of this.fantasmas) {
      f.asustado = false;
      f.color    = f.colorOriginal;
    }
  }

  /* â”€â”€ Umbrales y timer de inactividad â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Revisa si algÃºn fantasma en 'esperando' ya alcanzÃ³ su umbral
   * de puntos comidos y, si es asÃ­, inicia su animaciÃ³n de salida.
   */
  _verificarUmbralesFantasmas() {
    for (const f of this.fantasmas) {
      if (
        f.estadoSalida === 'esperando' &&
        this.puntosComidos >= f.umbralPuntos
      ) {
        f.estadoSalida = 'centrando';
      }
    }
  }

  /**
   * Si el timer de inactividad supera el umbral, fuerza la salida
   * del primer fantasma que siga en estado 'esperando'.
   */
  _verificarTimerInactividad() {
    if (this.ticksSinComer < this.TIMER_INACTIVIDAD) return;

    const siguiente = this.fantasmas.find(f => f.estadoSalida === 'esperando');
    if (siguiente) {
      siguiente.estadoSalida = 'centrando';
    }
    /* Reiniciar el timer para que no fuerce salidas en cascada */
    this.ticksSinComer = 0;
  }

  /* â”€â”€ AnimaciÃ³n de salida de la caja â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Procesa la animaciÃ³n de salida para un fantasma segÃºn su estado:
   *  'centrando' â†’ se desplaza a CAJA_CENTRO_COL
   *  'subiendo'  â†’ sube hasta PUERTA_FILA
   *  'libre'     â†’ sale de la funciÃ³n (se procesarÃ¡ con _moverFantasmaLibre)
   *
   * @param {Object} f - Fantasma
   */
  _procesarSalidaCaja(f) {
    if (f.estadoSalida === 'centrando') {
      if (f.col < CAJA_CENTRO_COL) {
        f.col++;
        f.direccion = { dc: 1, df: 0 };
      } else if (f.col > CAJA_CENTRO_COL) {
        f.col--;
        f.direccion = { dc: -1, df: 0 };
      } else {
        /* LlegÃ³ al centro: comenzar a subir */
        f.estadoSalida = 'subiendo';
        f.direccion    = { dc: 0, df: -1 };
      }
      return;
    }

    if (f.estadoSalida === 'subiendo') {
      if (f.fila > PUERTA_FILA) {
        f.fila--;
        f.direccion = { dc: 0, df: -1 };
      } else {
        /*
         * CruzÃ³ la puerta: queda libre.
         * SegÃºn la especificaciÃ³n, al salir el fantasma gira a la izquierda.
         */
        f.estadoSalida  = 'libre';
        f.fueraDeCaja   = true;
        f.direccion     = { dc: -1, df: 0 };  // girar a la izquierda
        f.dirProhibida  = { dc: 1,  df: 0 };  // no puede volver a la derecha inmediato
      }
    }
  }

  /* â”€â”€ Movimiento de fantasmas libres â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Mueve un fantasma libre usando IA de persecuciÃ³n / dispersiÃ³n.
   * El fantasma no puede dar media vuelta (direcciÃ³n prohibida) y
   * no puede superponerse con otro fantasma.
   *
   * @param {Object} f          - Fantasma a mover
   * @param {number} indice     - Ãndice en this.fantasmas (para anti-superposiciÃ³n)
   */
  _moverFantasmaLibre(f, indice) {
    f.ticksMovimiento++;

    /* Velocidad base: los fantasmas se mueven cada 2 ticks (~200 ms).
       Si estÃ¡n asustados, se mueven cada 3 ticks (mÃ¡s lentos). */
    const intervalo = f.asustado ? 3 : Math.max(1, 3 - Math.floor(this.nivel / 3));
    if (f.ticksMovimiento < intervalo) return;
    f.ticksMovimiento = 0;

    const direcciones = [
      { dc:  1, df:  0 },
      { dc: -1, df:  0 },
      { dc:  0, df:  1 },
      { dc:  0, df: -1 }
    ];

    /* Objetivo: perseguir al Pac-Man mÃ¡s cercano (o alejarse si asustado) */
    let objetivo = this.pacman1;
    if (this.pacman2 && this.pacman2.vivo) {
      const d1 = Math.abs(f.col - this.pacman1.col) + Math.abs(f.fila - this.pacman1.fila);
      const d2 = Math.abs(f.col - this.pacman2.col) + Math.abs(f.fila - this.pacman2.fila);
      if (d2 < d1) objetivo = this.pacman2;
    }

    /* Ordenar direcciones por distancia al objetivo */
    const dirOrdenadas = [...direcciones].sort((a, b) => {
      const distA = Math.abs(f.col + a.dc - objetivo.col) + Math.abs(f.fila + a.df - objetivo.fila);
      const distB = Math.abs(f.col + b.dc - objetivo.col) + Math.abs(f.fila + b.df - objetivo.fila);
      /* Si asustado, se aleja; si normal, persigue */
      return f.asustado ? distB - distA : distA - distB;
    });

    /* Elegir la primera direcciÃ³n vÃ¡lida:
       - camino libre para fantasmas
       - no es la direcciÃ³n prohibida (anti-rebote de 180Â°)
       - la celda destino no estÃ¡ ocupada por otro fantasma */
    for (const dir of dirOrdenadas) {
      const nc = f.col + dir.dc;
      const nf = f.fila + dir.df;

      /* Descartar direcciÃ³n prohibida */
      if (dir.dc === f.dirProhibida.dc && dir.df === f.dirProhibida.df) continue;

      /* Descartar si la celda no es transitable */
      if (!esCeldaLibreFantasma(this.mapa, nc, nf)) continue;

      /* Descartar si otro fantasma libre ya ocupa esa celda */
      const ocupadaPorOtro = this.fantasmas.some((otro, i) => {
        if (i === indice) return false;                 // no comparar consigo mismo
        if (otro.estadoSalida !== 'libre') return false; // solo considerar los libres
        return otro.col === nc && otro.fila === nf;
      });
      if (ocupadaPorOtro) continue;

      /* Mover */
      f.col          = nc;
      f.fila         = nf;
      f.dirProhibida = { dc: -dir.dc, df: -dir.df };  // guardar opuesto
      f.direccion    = dir;
      break;
    }

    /* Decrementar tiempo asustado */
    if (f.asustado) {
      f.tiempoAsustado--;
      if (f.tiempoAsustado <= 0) {
        f.asustado = false;
        f.color    = f.colorOriginal;
      }
    }
  }

  /* â”€â”€ Orquestador de fantasmas â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Procesa todos los fantasmas en cada tick:
   *  - Si aÃºn estÃ¡n en la caja (centrando/subiendo): animaciÃ³n de salida.
   *  - Si ya estÃ¡n libres: movimiento de IA.
   */
  _procesarFantasmas() {
    this.fantasmas.forEach((f, indice) => {
      if (f.estadoSalida === 'libre') {
        this._moverFantasmaLibre(f, indice);
      } else if (
        f.estadoSalida === 'centrando' ||
        f.estadoSalida === 'subiendo'
      ) {
        /*
         * La animaciÃ³n de salida se ejecuta cada 2 ticks
         * para que se vea suave y no sea instantÃ¡nea.
         */
        if (this.tick % 2 === 0) {
          this._procesarSalidaCaja(f);
        }
      }
      /* 'esperando' â†’ no hace nada hasta que se active */
    });
  }

  /* â”€â”€ Colisiones Pac-Man â†” Fantasmas â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Detecta si algÃºn Pac-Man estÃ¡ en la misma celda que un fantasma.
   * - Si el fantasma estÃ¡ asustado: Pac-Man lo come (200 pts) y
   *   el fantasma vuelve a la caja.
   * - Si no estÃ¡ asustado: Pac-Man pierde una vida.
   */
  _detectarColisiones() {
    const pacmans = this.pacman2
      ? [this.pacman1, this.pacman2]
      : [this.pacman1];

    for (const pm of pacmans) {
      if (!pm.vivo) continue;

      for (let i = 0; i < this.fantasmas.length; i++) {
        const f = this.fantasmas[i];

        /* Solo colisionar con fantasmas libres */
        if (f.estadoSalida !== 'libre') continue;

        if (f.col === pm.col && f.fila === pm.fila) {
          if (pm.poderActivo && f.asustado) {
            /* Pac-Man come al fantasma */
            pm.puntaje += 200;
            this._regresarFantasmaCaja(i);
          } else {
            /* Pac-Man pierde una vida */
            pm.vidas--;
            if (pm.vidas <= 0) {
              pm.vivo = false;
            } else {
              pm.col  = pm.colRespawn;
              pm.fila = pm.filaRespawn;
              pm.direccion        = { dc: 0, df: 0 };
              pm.proximaDireccion = { dc: 1, df: 0 };
            }
          }
        }
      }
    }
  }

  /**
   * Regresa un fantasma comido a su posiciÃ³n inicial dentro de la caja
   * y reinicia su estado de salida (debe volver a esperar/salir).
   * @param {number} indice - Ãndice del fantasma en this.fantasmas
   */
  _regresarFantasmaCaja(indice) {
    const def = DEFINICION_FANTASMAS[indice];
    const f   = this.fantasmas[indice];

    f.col           = def.colInicio === PUERTA_COL ? CAJA_CENTRO_COL : def.colInicio;
    f.fila          = def.filaInicio === PUERTA_FILA ? 10 : def.filaInicio;
    f.asustado      = false;
    f.color         = f.colorOriginal;
    f.fueraDeCaja   = false;
    f.estadoSalida  = 'centrando'; // vuelve a salir enseguida
    f.direccion     = { dc: 0, df: 0 };
    f.dirProhibida  = { dc: 0, df: 0 };
  }

  /* â”€â”€ Avanzar de nivel â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Reinicia el tablero para el siguiente nivel.
   * Aumenta la velocidad y reduce los umbrales de salida de los fantasmas.
   */
  _avanzarNivel() {
    this.nivel++;
    this.mapa          = clonarMapa();
    this.puntosRestantes = this._contarPuntos();
    this.puntosComidos  = 0;
    this.ticksSinComer  = 0;

    /* Reiniciar Pac-Men a su posiciÃ³n de respawn */
    this.pacman1.col  = this.pacman1.colRespawn;
    this.pacman1.fila = this.pacman1.filaRespawn;
    this.pacman1.direccion        = { dc: 0, df: 0 };
    this.pacman1.proximaDireccion = { dc: 1, df: 0 };
    this.pacman1.poderActivo      = false;

    if (this.pacman2) {
      this.pacman2.col  = this.pacman2.colRespawn;
      this.pacman2.fila = this.pacman2.filaRespawn;
      this.pacman2.direccion        = { dc: 0, df: 0 };
      this.pacman2.proximaDireccion = { dc: 1, df: 0 };
      this.pacman2.poderActivo      = false;
    }

    /* Recrear fantasmas y ajustar umbrales al nuevo nivel */
    this.fantasmas = DEFINICION_FANTASMAS.map(def => crearFantasma(def));
    this._ajustarUmbrales();
    this._iniciarSalidaInmediata();
  }

  /* â”€â”€ Renderizado â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Dibuja el estado completo del juego en el canvas:
   * fondo, mapa (paredes, puntos, power-ups), fantasmas y Pac-Men.
   */
  renderizar() {
    const ctx   = this.ctx;
    const ancho = this.canvas.width;
    const alto  = this.canvas.height;

    /* Fondo negro */
    ctx.fillStyle = '#000000';
    ctx.fillRect(0, 0, ancho, alto);

    /* Mapa */
    for (let f = 0; f < FILAS; f++) {
      for (let c = 0; c < COLUMNAS; c++) {
        const celda = this.mapa[f][c];
        const x     = c * CELDA;
        const y     = f * CELDA;

        if (celda === 1) {
          /* Pared azul con borde oscuro */
          ctx.fillStyle   = '#1a1aff';
          ctx.fillRect(x, y, CELDA, CELDA);
          ctx.strokeStyle = '#0000aa';
          ctx.lineWidth   = 1;
          ctx.strokeRect(x + 1, y + 1, CELDA - 2, CELDA - 2);

        } else if (celda === 0) {
          /* Punto pequeÃ±o */
          ctx.fillStyle = '#ffddaa';
          ctx.beginPath();
          ctx.arc(x + CELDA / 2, y + CELDA / 2, 2, 0, Math.PI * 2);
          ctx.fill();

        } else if (celda === 3) {
          /* Power-up parpadeante */
          if (Math.floor(this.tick / 5) % 2 === 0) {
            ctx.fillStyle    = '#ffffff';
            ctx.shadowColor  = '#ffffff';
            ctx.shadowBlur   = 8;
            ctx.beginPath();
            ctx.arc(x + CELDA / 2, y + CELDA / 2, 5, 0, Math.PI * 2);
            ctx.fill();
            ctx.shadowBlur = 0;
          }
        }
        /* celdas 2 y 4 â†’ se dibujan vacÃ­as (fondo negro) */
      }
    }

    /* Dibujar fantasmas (primero, debajo de Pac-Man) */
    this.fantasmas.forEach(f => this._dibujarFantasma(ctx, f));

    /* Dibujar Pac-Men */
    this._dibujarPacman(ctx, this.pacman1);
    if (this.pacman2) this._dibujarPacman(ctx, this.pacman2);
  }

  /* â”€â”€ Dibujo de personajes â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€ */

  /**
   * Dibuja un Pac-Man con animaciÃ³n de apertura de boca.
   * La boca apunta en la direcciÃ³n de movimiento.
   * @param {CanvasRenderingContext2D} ctx
   * @param {Object} pm
   */
  _dibujarPacman(ctx, pm) {
    if (!pm.vivo) return;

    const x     = pm.col * CELDA + CELDA / 2;
    const y     = pm.fila * CELDA + CELDA / 2;
    const radio = CELDA / 2 - 1;

    /* Ãngulo base segÃºn la direcciÃ³n actual */
    let angBase = 0;
    if      (pm.direccion.dc === -1) angBase = Math.PI;
    else if (pm.direccion.df ===  1) angBase = Math.PI / 2;
    else if (pm.direccion.df === -1) angBase = -Math.PI / 2;

    const ap = pm.anguloApertura;

    ctx.fillStyle   = pm.color;
    ctx.shadowColor = pm.color;
    ctx.shadowBlur  = 6;
    ctx.beginPath();
    ctx.moveTo(x, y);
    ctx.arc(x, y, radio, angBase + ap, angBase + Math.PI * 2 - ap);
    ctx.closePath();
    ctx.fill();
    ctx.shadowBlur = 0;
  }

  /**
   * Dibuja un fantasma.
   * - Color normal si estÃ¡ libre y no asustado.
   * - Azul si estÃ¡ asustado.
   * - Parpadeo blanco/azul cuando el poder estÃ¡ a punto de terminar.
   * - Semitransparente mientras sale de la caja (centrando/subiendo).
   * @param {CanvasRenderingContext2D} ctx
   * @param {Object} f
   */
  _dibujarFantasma(ctx, f) {
    const x     = f.col  * CELDA;
    const y     = f.fila * CELDA;
    const radio = CELDA / 2 - 1;

    /* Determinar color */
    let colorRelleno;
    if (f.asustado) {
      const parpadeando = f.tiempoAsustado < 15 && Math.floor(this.tick / 3) % 2 === 0;
      colorRelleno = parpadeando ? '#ffffff' : '#2222ff';
    } else {
      colorRelleno = f.color;
    }

    /* Opacidad reducida mientras estÃ¡ saliendo de la caja */
    ctx.globalAlpha = (f.estadoSalida === 'libre') ? 1.0 : 0.6;

    ctx.fillStyle   = colorRelleno;
    ctx.shadowColor = colorRelleno;
    ctx.shadowBlur  = 5;

    /* Cuerpo: semicÃ­rculo superior */
    ctx.beginPath();
    ctx.arc(x + CELDA / 2, y + radio + 1, radio, Math.PI, 0);
    ctx.lineTo(x + CELDA - 1, y + CELDA - 3);

    /* FaldÃ³n ondulado en la parte inferior */
    const numOndas  = 3;
    const anchoOnda = (CELDA - 2) / numOndas;
    for (let i = numOndas; i >= 0; i--) {
      const xBase = x + 1 + i * anchoOnda - anchoOnda / 2;
      ctx.arc(xBase, y + CELDA - 3, anchoOnda / 2, 0, Math.PI, i % 2 === 0);
    }
    ctx.lineTo(x + 1, y + radio + 1);
    ctx.closePath();
    ctx.fill();
    ctx.shadowBlur = 0;

    /* Ojos */
    if (!f.asustado) {
      /* Ojos blancos */
      ctx.fillStyle = '#ffffff';
      ctx.beginPath();
      ctx.arc(x + CELDA / 2 - 4, y + radio, 3, 0, Math.PI * 2);
      ctx.arc(x + CELDA / 2 + 4, y + radio, 3, 0, Math.PI * 2);
      ctx.fill();
      /* Pupilas azul oscuro */
      ctx.fillStyle = '#000088';
      ctx.beginPath();
      ctx.arc(x + CELDA / 2 - 3, y + radio, 1.5, 0, Math.PI * 2);
      ctx.arc(x + CELDA / 2 + 5, y + radio, 1.5, 0, Math.PI * 2);
      ctx.fill();
    } else {
      /* Ojos asustados */
      ctx.fillStyle = '#ffffff';
      ctx.beginPath();
      ctx.arc(x + CELDA / 2 - 3, y + radio, 2, 0, Math.PI * 2);
      ctx.arc(x + CELDA / 2 + 3, y + radio, 2, 0, Math.PI * 2);
      ctx.fill();
    }

    /* Restaurar opacidad */
    ctx.globalAlpha = 1.0;
  }
}

function registrarBotonTactil(idBoton, jugador, dir) {
  const btn = document.getElementById(idBoton);
  if (!btn) return;
  const handler = (e) => {
    e.preventDefault();

/**
 * Rutas Express para CRUD completo de jugadores del juego Pac-Man.
 * Persiste los datos en /data/jugadores.txt usando fs (ES Modules).
 */

import express from "express";
import fs from "fs";
import path from "path";
import { fileURLToPath } from "url";

const router = express.Router();
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const rutaArchivo = path.join(__dirname, "../data/jugadores.txt");

/**
 * Asegura que el directorio /data y el archivo .txt existan.
 */
function asegurarArchivo() {
  const dir = path.join(__dirname, "../data");
  if (!fs.existsSync(dir)) {
    fs.mkdirSync(dir, { recursive: true });
  }
  if (!fs.existsSync(rutaArchivo)) {
    fs.writeFileSync(rutaArchivo, JSON.stringify([]), "utf-8");
  }
}

/**
 * Lee y parsea la lista de jugadores desde el archivo .txt.
 * @returns {Array}
 */
function leerJugadores() {
  asegurarArchivo();
  try {
    return JSON.parse(fs.readFileSync(rutaArchivo, "utf-8"));
  } catch {
    return [];
  }
}

/**
 * Escribe la lista de jugadores en el archivo .txt.
 * @param {Array} jugadores
 */
function escribirJugadores(jugadores) {
  asegurarArchivo();
  fs.writeFileSync(rutaArchivo, JSON.stringify(jugadores, null, 2), "utf-8");
}

/* GET: obtener todos los jugadores */
router.get("/", (req, res) => {
  res.json(leerJugadores());
});

/* GET: descargar archivo .txt */
router.get("/archivo", (req, res) => {
  asegurarArchivo();
  res.download(rutaArchivo, "jugadores_pacman.txt");
});

/**
 * POST: agregar jugador nuevo.
 * Recibe: { nombre, puntaje, modo, nivel }
 */
router.post("/", (req, res) => {
  const { nombre, puntaje, modo, nivel } = req.body;
  if (!nombre || puntaje === undefined || !modo) {
    return res.status(400).json({ error: "Datos incompletos" });
  }

  const jugadores = leerJugadores();
  const nuevoJugador = {
    id: Date.now().toString(),
    nombre: nombre.trim(),
    puntaje: Number(puntaje),
    modo,
    nivel: nivel || 1,
    fecha: new Date().toLocaleDateString("es-AR")
  };

  jugadores.push(nuevoJugador);
  escribirJugadores(jugadores);
  res.status(201).json(nuevoJugador);
});

/**
 * PUT: modificar jugador por ID.
 */
router.put("/:id", (req, res) => {
  const { id } = req.params;
  const { nombre, puntaje, modo } = req.body;
  const jugadores = leerJugadores();
  const idx = jugadores.findIndex((j) => j.id === id);

  if (idx === -1) return res.status(404).json({ error: "No encontrado" });

  if (nombre !== undefined) jugadores[idx].nombre = nombre.trim();
  if (puntaje !== undefined) jugadores[idx].puntaje = Number(puntaje);
  if (modo !== undefined) jugadores[idx].modo = modo;

  escribirJugadores(jugadores);
  res.json(jugadores[idx]);
});

/**
 * DELETE: eliminar jugador por ID.
 */
router.delete("/:id", (req, res) => {
  const { id } = req.params;
  const jugadores = leerJugadores();
  const nueva = jugadores.filter((j) => j.id !== id);

  if (nueva.length === jugadores.length) {
    return res.status(404).json({ error: "No encontrado" });
  }

  escribirJugadores(nueva);
  res.json({ mensaje: "Jugador eliminado correctamente" });
});

export default router;
